-- Funktionalität hinzufügen
local function toggleButton(button, enabledText, disabledText)
    local isEnabled = button.BackgroundColor3 == Color3.fromRGB(50, 255, 50)
    
    if isEnabled then
        button.BackgroundColor3 = Color3.fromRGB(255, 50, 50)
        button.Text = disabledText
    else
        button.BackgroundColor3 = Color3.fromRGB(50, 255, 50)
        button.Text = enabledText
    end
    
    return not isEnabled
end

-- Tab-Wechsel-Funktionalität
local function showTab(tabName)
    AimbotContent.Visible = false
    VisualsContent.Visible = false
    ExploitsContent.Visible = false
    SettingsContent.Visible = false
    
    AimbotTab.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    VisualsTab.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    ExploitsTab.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    SettingsTab.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    
    if tabName == "Aimbot" then
        AimbotContent.Visible = true
        AimbotTab.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    elseif tabName == "Visuals" then
        VisualsContent.Visible = true
        VisualsTab.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    elseif tabName == "Exploits" then
        ExploitsContent.Visible = true
        ExploitsTab.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    elseif tabName == "Settings" then
        SettingsContent.Visible = true
        SettingsTab.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    end
end

-- Slider-Funktionalität
local function makeSliderDraggable(slider)
    local knob = slider:FindFirstChild(slider.Name .. "Knob")
    local bar = slider:FindFirstChild(slider.Name .. "Bar")
    local valueLabel = slider:FindFirstChild(slider.Name .. "Value")
    
    local dragging = false
    local function updateValue(input)
        local relativePosX = math.clamp((input.Position.X - bar.AbsolutePosition.X) / bar.AbsoluteSize.X, 0, 1)
        knob.Position = UDim2.new(relativePosX, -5, 0, -5)
        
        -- Werte je nach Slider anpassen
        if slider.Name == "FOVSlider" then
            local fov = math.floor(relativePosX * 180)
            valueLabel.Text = tostring(fov)
            -- Hier FOV-Funktionalität implementieren
        elseif slider.Name == "SmoothingSlider" then
            local smoothing = math.floor(relativePosX * 10)
            valueLabel.Text = tostring(smoothing)
            -- Hier Smoothing-Funktionalität implementieren
        elseif slider.Name == "SpeedSlider" then
            local speed = 1 + (relativePosX * 4)
            valueLabel.Text = string.format("%.1fx", speed)
            -- Hier Speed-Funktionalität implementieren
        elseif slider.Name == "HitboxSlider" then
            local hitbox = 1 + (relativePosX * 4)
            valueLabel.Text = string.format("%.1fx", hitbox)
            -- Hier Hitbox-Funktionalität implementieren
        end
    end
    
    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)
    
    knob.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    bar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            updateValue(input)
            dragging = true
        end
    end)
    
    bar.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            updateValue(input)
        end
    end)
end
